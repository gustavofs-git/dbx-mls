"""
Unit tests for src/riot_client.py.

Coverage targets:
  - RiotRateLimiter: token bucket initialization, acquire() consumes tokens, refill logic
  - call_riot_api: normal 200, 429 application limit, 429 service limit, 404 RiotApiError
  - Retry-After sleep honored exactly
  - Both buckets decremented per call
"""
import pytest
from src.riot_client import RiotRateLimiter, call_riot_api
from src.common.exceptions import RiotApiError


class TestRiotRateLimiter:
    """Token bucket initialization and acquire() behavior."""

    def test_initial_bucket_values(self):
        """Both buckets start at full capacity."""
        limiter = RiotRateLimiter()
        assert limiter._sec_capacity == 20
        assert limiter._min_capacity == 100
        assert limiter._sec_tokens == 20
        assert limiter._min_tokens == 100

    def test_refill_rate_constants(self):
        """Refill rates match Riot API limits."""
        limiter = RiotRateLimiter()
        assert limiter._sec_refill_rate == 20.0
        assert abs(limiter._min_refill_rate - (100 / 120.0)) < 1e-9

    def test_acquire_decrements_both_buckets(self):
        """acquire() consumes one token from each bucket."""
        limiter = RiotRateLimiter()
        initial_sec = limiter._sec_tokens
        initial_min = limiter._min_tokens
        limiter.acquire()
        assert limiter._sec_tokens < initial_sec
        assert limiter._min_tokens < initial_min

    def test_acquire_returns_when_tokens_available(self, mocker):
        """acquire() returns immediately when both buckets have tokens."""
        limiter = RiotRateLimiter()
        mock_sleep = mocker.patch("time.sleep")
        limiter.acquire()
        # No sleep needed when tokens are available
        mock_sleep.assert_not_called()

    def test_acquire_sleeps_when_sec_bucket_empty(self, mocker):
        """acquire() sleeps (0.05s lock-release wait) when per-second bucket is depleted."""
        limiter = RiotRateLimiter()
        limiter._sec_tokens = 0.0  # deplete the per-second bucket

        # Patch time.sleep and have _refill restore tokens on second call
        call_count = [0]

        def fake_sleep(t):
            call_count[0] += 1
            if call_count[0] >= 1:
                # Simulate time passing — refill buckets
                limiter._sec_tokens = 20.0

        mocker.patch("time.sleep", side_effect=fake_sleep)
        limiter.acquire()
        assert call_count[0] >= 1


class TestCallRiotApi:
    """call_riot_api() HTTP handling tests."""

    def test_normal_200_response(self, mocker, mock_response_200):
        """Normal 200 response returns parsed JSON."""
        payload = {"matchId": "KR_123456789"}
        mocker.patch("requests.get", return_value=mock_response_200(payload))
        limiter = RiotRateLimiter()
        result = call_riot_api("https://kr.api.riotgames.com/test", {}, limiter)
        assert result == payload

    def test_calls_limiter_acquire_before_request(self, mocker, mock_response_200):
        """limiter.acquire() is called exactly once before requests.get."""
        mocker.patch("requests.get", return_value=mock_response_200({}))
        limiter = RiotRateLimiter()
        acquire_spy = mocker.spy(limiter, "acquire")
        call_riot_api("https://test.example.com", {}, limiter)
        acquire_spy.assert_called_once()

    def test_404_raises_riot_api_error(self, mocker):
        """HTTP 404 raises RiotApiError with correct status_code and url."""
        mock_404 = mocker.Mock()
        mock_404.status_code = 404
        mock_404.headers = {}
        mocker.patch("requests.get", return_value=mock_404)
        limiter = RiotRateLimiter()
        with pytest.raises(RiotApiError) as exc_info:
            call_riot_api("https://test.example.com/not-found", {}, limiter)
        assert exc_info.value.status_code == 404
        assert "not-found" in exc_info.value.url

    def test_429_application_limit_sleeps_retry_after(self, mocker, mock_response_200, mock_response_429):
        """429 with X-Rate-Limit-Type: application sleeps Retry-After seconds then retries."""
        mock_sleep = mocker.patch("time.sleep")
        mocker.patch(
            "requests.get",
            side_effect=[mock_response_429("application", "3"), mock_response_200({"ok": True})]
        )
        limiter = RiotRateLimiter()
        result = call_riot_api("https://test.example.com", {}, limiter)
        assert result == {"ok": True}
        # Verify sleep was called with 3 (the Retry-After value)
        sleep_calls = [c.args[0] for c in mock_sleep.call_args_list]
        assert 3 in sleep_calls

    def test_429_method_limit_sleeps_retry_after(self, mocker, mock_response_200, mock_response_429):
        """429 with X-Rate-Limit-Type: method also sleeps Retry-After and retries."""
        mock_sleep = mocker.patch("time.sleep")
        mocker.patch(
            "requests.get",
            side_effect=[mock_response_429("method", "2"), mock_response_200({})]
        )
        limiter = RiotRateLimiter()
        call_riot_api("https://test.example.com", {}, limiter)
        sleep_calls = [c.args[0] for c in mock_sleep.call_args_list]
        assert 2 in sleep_calls

    def test_429_service_limit_triggers_backoff(self, mocker, mock_response_200, mock_response_429):
        """429 with X-Rate-Limit-Type: service triggers exponential backoff (sleep > 0)."""
        mock_sleep = mocker.patch("time.sleep")
        mocker.patch(
            "requests.get",
            side_effect=[mock_response_429("service", "1"), mock_response_200({})]
        )
        limiter = RiotRateLimiter()
        call_riot_api("https://test.example.com", {}, limiter)
        # Service limit uses backoff — sleep must have been called with value >= 2
        sleep_calls = [c.args[0] for c in mock_sleep.call_args_list]
        assert any(s >= 2 for s in sleep_calls)

    def test_default_retry_after_is_1_when_header_missing(self, mocker, mock_response_200):
        """If Retry-After header is absent on 429, defaults to 1 second sleep."""
        mock_sleep = mocker.patch("time.sleep")
        mock_429 = mocker.Mock()
        mock_429.status_code = 429
        mock_429.headers = {"X-Rate-Limit-Type": "application"}  # no Retry-After header
        mocker.patch("requests.get", side_effect=[mock_429, mock_response_200({})])
        limiter = RiotRateLimiter()
        call_riot_api("https://test.example.com", {}, limiter)
        sleep_calls = [c.args[0] for c in mock_sleep.call_args_list]
        assert 1 in sleep_calls

**`databricks.yml`** — root bundle definition:

```yaml
bundle:
  name: dbx-mls
  databricks_cli_version: ">=0.250.0"

include:
  - resources/*.yml
  - resources/jobs/*.yml

variables:
  databricks_host:
    description: "Databricks workspace URL (e.g. https://adb-xxxx.azuredatabricks.net)"
  sp_client_id:
    description: "Service Principal client ID (UUID) used for prod run_as"

targets:
  dev:
    mode: development
    default: true
    workspace:
      host: ${var.databricks_host}

  prod:
    mode: production
    workspace:
      host: ${var.databricks_host}
      root_path: /Workspace/Shared/.bundle/dbx-mls/prod
    run_as:
      service_principal_name: ${var.sp_client_id}
```

**`resources/clusters.yml`** — shared job cluster (DBR 16.4 LTS, Standard_F4s_v2, per D-01, D-02, D-03):

```yaml
# Shared job cluster definition used by all DAB jobs.
# node_type_id: Standard_F4s_v2 — 8 GB RAM, 4 vCPUs; ~30% cheaper than Standard_DS3_v2
# autotermination_minutes: 0 — correct for job clusters; they terminate when the job finishes
# spark_version: 16.4.x-scala2.12 — Databricks Runtime 16.4 LTS
# spark.databricks.cluster.profile: singleNode — this is the correct profile value for a single-node
#   job cluster (num_workers: 0). Do NOT use 'singleUser'; that is a Unity Catalog access mode,
#   not a cluster profile name.

resources:
  clusters:
    job_cluster_shared:
      cluster_name: dbx-mls-job-cluster
      spark_version: 16.4.x-scala2.12
      node_type_id: Standard_F4s_v2
      num_workers: 0
      spark_conf:
        spark.databricks.cluster.profile: singleNode
        spark.master: local[*]
      azure_attributes:
        first_on_demand: 1
        availability: ON_DEMAND_AZURE
      autotermination_minutes: 0
      custom_tags:
        project: dbx-mls
        phase: infrastructure
```

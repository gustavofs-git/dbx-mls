Actual ci.yml env block and steps (relevant portion):
```yaml
permissions:
  id-token: write   # REQUIRED for OIDC token issuance
  contents: read

jobs:
  validate-and-test:
    runs-on: ubuntu-latest
    env:
      DATABRICKS_AUTH_TYPE: azure-cli
      DATABRICKS_HOST: ${{ vars.DATABRICKS_HOST }}
      BUNDLE_VAR_sp_client_id: ${{ vars.DATABRICKS_CLIENT_ID }}
    steps:
      - uses: actions/checkout@v4
      - uses: azure/login@v2
        with:
          client-id: ${{ vars.DATABRICKS_CLIENT_ID }}
          tenant-id: ${{ vars.AZURE_TENANT_ID }}
          allow-no-subscriptions: true
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - run: pip install -r requirements-dev.txt
      - uses: databricks/setup-cli@v0.295.0
      - run: databricks bundle validate
      - run: pytest tests/unit/ --cov=src --cov-report=xml
```

Azure AD federation: configured as Federated Identity Credential on the App Registration in Azure portal (or via `az ad app federated-credential create` Azure CLI). NOT via `databricks account service-principal-federation-policy create`.

Three GitHub repository-level variables used: DATABRICKS_HOST, DATABRICKS_CLIENT_ID, AZURE_TENANT_ID.
